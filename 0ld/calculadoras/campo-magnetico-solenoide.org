<script>
function calcular(){
    l = document.getElementById("longitud").value;
    n = document.getElementById("espiras").value;
    i = document.getElementById("corriente").value;
    mu0 = document.getElementById("magnetica").value;
    mur = document.getElementById("relativa").value;

    formato = document.querySelector('input[name="format"]:checked').value;

switch(formato){
  case 'decimal':
    resultado = nerdamer(' (muo*mur*i*n)/(l) ', {
        muo: mu0,
        mur: mur,
        i: i,
        n: n,
        l: l,
       pi: Math.PI
      });
    document.getElementById('resultado').innerHTML = resultado.toDecimal();
  break;

  case 'fraccion':
    resultado = nerdamer(' (muo*mur*i*n)/(l) ', {
        muo: mu0,
        mur: mur,
        i: i,
        n: n,
        l: l
      });
    document.getElementById('resultado').innerHTML = "$${ " + resultado.toString() + "}$$ ";
  break;
}

    MathJax.typeset()
    nerdamer.clearVars();
}

</script>
