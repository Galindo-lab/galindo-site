function calcular() {

        p1: "vector(" + getById("p_1").value + ")",
        p2: "vector(" + getById("p_2").value + ")",

    case "fraccion":
    getById('resultado').innerHTML = "$${ "  + resultado.toString() + " }$$" ;
    break;
}
